void shuffle(Random random) {
  List<Ball> balls = Ball.values.toList()..shuffle(random);
  List<Ring> rings = Ring.values.toList()..shuffle(random);
  for (int i = 0; i < balls.length; i++) {
    Ball ball = balls[i];
    Ring ring = rings[i];
    _ringWith[ball] = ring;
    _ballWith[ring] = ball;
  }
}
